git init : khởi tạo repo local
git remote add origin https://github.com/tuananhhdev/demo_git.git liên kết repo local và repo remote

+ đẩy code từ local lên remote : 3 bước
git add-A : chọn
git commit -m "tao ra trang home" : xác nhận và ghi kèm chú thích
git push -u origin master : đẩy lần đầu
git push : lần 2 trở đi 

+ setup account
git config user.name "tuananh"
git config user.email "tuananh@gmail.com"

branch : nhánh
git checkout -b login : tạo ra nhánh login từ nhánh hiện đại 
git branch : list các branch hiện có

git merge login : đưa code từ nhánh login về nhánh hiện tại 


+ Lấy source từ repo
git clone ủrl . : clone source vào folder hiện tại 
git pull : lấy code mới nhất từ remote


git log : xem danh sách lệnh đã commit

git stash : khôi phục code về lần commit gần nhất
